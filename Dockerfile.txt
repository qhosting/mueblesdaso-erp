# Dockerfile optimizado para Easypanel - Mueblesdaso ERP
# Este archivo sirve la aplicación como una PWA estática de alto rendimiento.

FROM nginx:stable-alpine

# Etiquetado para identificación en Easypanel
LABEL org.opencontainers.image.title="Mueblesdaso ERP"
LABEL org.opencontainers.image.description="Sistema ERP/CRM para venta a crédito"
LABEL org.opencontainers.image.version="2.1.0"
LABEL org.opencontainers.image.vendor="Mueblesdaso"

# Limpiar el directorio predeterminado de Nginx
RUN rm -rf /usr/share/nginx/html/*

# Copiar los archivos de la aplicación al contenedor
# Nota: Se copian todos los archivos incluyendo index.tsx y componentes
# para que el navegador los procese vía ES Modules (según la arquitectura del proyecto).
COPY . /usr/share/nginx/html

# Copiar la configuración personalizada de Nginx
# Esta configuración es CRÍTICA porque define los MIME types para .tsx y el soporte SPA.
COPY nginx.conf /etc/nginx/conf.d/default.conf

# Ajustar permisos de usuario para seguridad
RUN chown -R nginx:nginx /usr/share/nginx/html && \
    chmod -R 755 /usr/share/nginx/html

# Healthcheck para monitoreo automático de Easypanel
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD wget --quiet --tries=1 --spider http://localhost/ || exit 1

# Exponer el puerto estándar de HTTP
EXPOSE 80

# Iniciar Nginx
CMD ["nginx", "-g", "daemon off;"]
